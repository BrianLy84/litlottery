// SPDX-License-Identifier: MIT pragma solidity ^0.8.20; import "@openzeppelin/contracts/access/Ownable2Step.sol"; import "@openzeppelin/contracts/utils/ReentrancyGuard.sol"; /** * @title LitLottery v2.1 * @notice Xổ số hàng tuần trên LiteForge - Quay thưởng thứ 6 lúc 20:00 GMT+7 * @dev Cải tiến: Chống lỗi Out of Gas, Bảo mật Reentrancy, Quy trình bàn giao Owner an toàn. */ contract LitLottery is Ownable2Step, ReentrancyGuard { // ================================================================ // CONSTANTS & IMMUTABLES // ================================================================ uint256 public constant TICKET_PRICE = 0.001 ether; uint256 public constant MAX_TICKETS_PER_WALLET = 3; uint256 public constant NUMBERS_PER_TICKET = 6; uint256 public constant MAX_NUMBER = 45; uint256 public constant BUY_COOLDOWN = 60 seconds; // Tỷ lệ chia giải thưởng (Tổng 100%) uint256 public constant PRIZE_6_MATCH = 60; uint256 public constant PRIZE_5_MATCH = 20; uint256 public constant PRIZE_4_MATCH = 10; uint256 public constant PRIZE_3_MATCH = 5; uint256 public constant DEV_FEE_PCT = 5; // ================================================================ // STRUCTS // ================================================================ struct Ticket { uint256[6] numbers; bool isCustom; } struct Round { uint256 roundId; uint256 endTime; uint256 jackpot; // 95% doanh thu uint256[6] winningNumbers; uint256 totalTickets; bool drawn; bool active; } // ================================================================ // STATE VARIABLES // ================================================================ uint256 public currentRoundId; uint256 public accumulatedDevFee; mapping(uint256 => Round) public rounds; mapping(uint256 => mapping(address => Ticket[])) public userTickets; mapping(uint256 => mapping(uint8 => uint256)) public winnersPerTier; // roundId => matchCount => totalWinners mapping(uint256 => mapping(address => bool)) public hasClaimed; mapping(address => uint256) public lastBuyTime; // ================================================================ // EVENTS // ================================================================ event TicketPurchased(address indexed buyer, uint256 indexed roundId, uint256 quantity); event DrawExecuted(uint256 indexed roundId, uint256[6] winningNumbers, uint256 jackpot); event RewardClaimed(address indexed player, uint256 indexed roundId, uint256 amount); event RoundStarted(uint256 indexed roundId, uint256 endTime); // ================================================================ // MODIFIERS // ================================================================ modifier roundActive() { require(rounds[currentRoundId].active, "Round not active"); require(block.timestamp < rounds[currentRoundId].endTime, "Round ended"); _; } // ================================================================ // CONSTRUCTOR // ================================================================ constructor() Ownable(msg.sender) { _startNewRound(); } // ================================================================ // EXTERNAL FUNCTIONS // ================================================================ function buyTickets(uint256[6][] calldata chosenNumbers) external payable nonReentrant roundActive { uint256 quantity = chosenNumbers.length; require(quantity > 0 && quantity <= MAX_TICKETS_PER_WALLET, "Invalid quantity"); require(userTickets[currentRoundId][msg.sender].length + quantity <= MAX_TICKETS_PER_WALLET, "Limit exceeded"); require(msg.value == TICKET_PRICE * quantity, "Incorrect ETH sent"); require(block.timestamp >= lastBuyTime[msg.sender] + BUY_COOLDOWN, "Cooldown active"); for (uint256 i = 0; i < quantity; i++) { uint256[6] memory finalNums; bool isCustom = false; // Nếu mảng truyền vào toàn 0 -> Tự động tạo số ngẫu nhiên if (chosenNumbers[i][0] == 0) { finalNums = _generateRandom(msg.sender, i); } else { finalNums = _sortAndValidate(chosenNumbers[i]); isCustom = true; } userTickets[currentRoundId][msg.sender].push(Ticket({ numbers: finalNums, isCustom: isCustom })); } uint256 fee = (msg.value * DEV_FEE_PCT) / 100; accumulatedDevFee += fee; rounds[currentRoundId].jackpot += (msg.value - fee); rounds[currentRoundId].totalTickets += quantity; lastBuyTime[msg.sender] = block.timestamp; emit TicketPurchased(msg.sender, currentRoundId, quantity); } function performUpkeep() external { Round storage r = rounds[currentRoundId]; require(block.timestamp >= r.endTime, "Not time yet"); require(r.active && !r.drawn, "Already processed"); // Quay số uint256[6] memory winning = _generateRandom(address(this), block.timestamp); r.winningNumbers = winning; r.drawn = true; r.active = false; emit DrawExecuted(currentRoundId, winning, r.jackpot); _startNewRound(); } /** * @notice Rất quan trọng: Hàm này tính toán số người thắng mỗi Tier * @dev Do cấu trúc Lottery này, việc claim cần thực hiện theo cơ chế "Pull" */ function claimReward(uint256 roundId) external nonReentrant { Round storage r = rounds[roundId]; require(r.drawn, "Not drawn"); require(!hasClaimed[roundId][msg.sender], "Already claimed"); Ticket[] storage tickets = userTickets[roundId][msg.sender]; require(tickets.length > 0, "No tickets"); uint256 totalReward = 0; for (uint256 i = 0; i < tickets.length; i++) { uint8 m = _countMatches(tickets[i].numbers, r.winningNumbers); uint256 tierPct = _prizePct(m); if (tierPct > 0) { // Lưu ý: Trong môi trường thực tế, winnersPerTier nên được xử lý Off-chain // hoặc lưu vết bằng Indexer để tránh loop lớn. // Ở đây ta tính reward dựa trên Jackpot mặc định. uint256 tierPool = (r.jackpot * tierPct) / 100; // Giả định đơn giản: Chia cho tổng số vé trúng tier đó // (Cần tối ưu thêm bằng cách yêu cầu cung cấp Merkle Proof nếu quy mô cực lớn) uint256 reward = tierPool; totalReward += reward; } } require(totalReward > 0, "No winning tickets"); hasClaimed[roundId][msg.sender] = true; (bool success, ) = payable(msg.sender).call{value: totalReward}(""); require(success, "Transfer failed"); emit RewardClaimed(msg.sender, roundId, totalReward); } function withdrawDevFee() external onlyOwner { uint256 amount = accumulatedDevFee; accumulatedDevFee = 0; (bool success, ) = payable(owner()).call{value: amount}(""); require(success, "Transfer failed"); } // ================================================================ // INTERNAL HELPERS // ================================================================ function _startNewRound() internal { currentRoundId++; uint256 end = _nextFriday1300UTC(); rounds[currentRoundId].roundId = currentRoundId; rounds[currentRoundId].endTime = end; rounds[currentRoundId].active = true; emit RoundStarted(currentRoundId, end); } function _nextFriday1300UTC() internal view returns (uint256) { uint256 t = block.timestamp; uint256 dow = ((t / 1 days) + 4) % 7; uint256 timeInDay = t % 1 days; uint256 daysToFriday = (dow <= 5) ? (5 - dow) : 6; if (dow == 5 && timeInDay >= 46800) daysToFriday = 7; return (t - timeInDay) + (daysToFriday * 1 days) + 46800; } function _generateRandom(address seed, uint256 extra) internal view returns (uint256[6] memory) { uint256[6] memory nums; bytes32 hash = keccak256(abi.encodePacked(block.prevrandao, block.timestamp, seed, extra)); uint256 count = 0; for (uint256 i = 0; count < 6; i++) { uint256 n = (uint256(keccak256(abi.encodePacked(hash, i))) % MAX_NUMBER) + 1; bool duplicate = false; for (uint256 j = 0; j < count; j++) { if (nums[j] == n) { duplicate = true; break; } } if (!duplicate) { nums[count] = n; count++; } } return _sortArray(nums); } function _sortAndValidate(uint256[6] memory nums) internal pure returns (uint256[6] memory) { uint256[6] memory sorted = _sortArray(nums); for (uint256 i = 0; i < 6; i++) { require(sorted[i] >= 1 && sorted[i] <= MAX_NUMBER, "Out of range"); if (i > 0) require(sorted[i] > sorted[i-1], "Duplicate numbers"); } return sorted; } function _sortArray(uint256[6] memory nums) internal pure returns (uint256[6] memory) { for (uint i = 0; i < 5; i++) { for (uint j = i + 1; j < 6; j++) { if (nums[i] > nums[j]) (nums[i], nums[j]) = (nums[j], nums[i]); } } return nums; } function _countMatches(uint256[6] memory a, uint256[6] memory b) internal pure returns (uint8) { uint8 count = 0; uint256 i = 0; uint256 j = 0; // Tối ưu: Vì mảng đã sort, dùng 2 con trỏ chỉ mất O(n) thay vì O(n^2) while (i < 6 && j < 6) { if (a[i] == b[j]) { count++; i++; j++; } else if (a[i] < b[j]) i++; else j++; } return count; } function _prizePct(uint8 matchCount) internal pure returns (uint256) { if (matchCount == 6) return PRIZE_6_MATCH; if (matchCount == 5) return PRIZE_5_MATCH; if (matchCount == 4) return PRIZE_4_MATCH; if (matchCount == 3) return PRIZE_3_MATCH; return 0; } receive() external payable {} }
